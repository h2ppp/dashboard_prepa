        // --- METTRE À JOUR LES DONNÉES DEPUIS L'ÉDITEUR ---

        function updateDataFromEditor() {

            const editor = document.getElementById('dataEditor');

            const text = editor.innerText;


            try {

                // Extraction des tableaux via eval (safe car environnement contrôlé)

                // On extrait ce qui est après "const jours = " et avant le ";"

                const joursMatch = text.match(/jours\s*=\s*\[([^\]]*)\]/);

                const dureesMatch = text.match(/durees\s*=\s*\[([^\]]*)\]/);

                const objMatch = text.match(/objectifQuotidien\s*=\s*([\d.]+)/);

                const matieresMatch = text.match(/matieres\s*=\s*\[([^\]]*)\]/);

                const tempsMatieresMatch = text.match(/tempsMatieres\s*=\s*\[([^\]]*)\]/);


                if (joursMatch) {

                    const raw = joursMatch[1].split(',').map(s => s.trim().replace(/['"]/g, ''));

                    if (raw.length === 7) jours = raw;

                }

                if (dureesMatch) {

                    const raw = dureesMatch[1].split(',').map(s => parseFloat(s.trim()));

                    if (raw.every(v => !isNaN(v)) && raw.length === 7) durees = raw;

                }

                if (objMatch) {

                    const val = parseFloat(objMatch[1]);

                    if (!isNaN(val) && val > 0) objectifQuotidien = val;

                }

                if (matieresMatch) {

                    const raw = matieresMatch[1].split(',').map(s => s.trim().replace(/['"]/g, ''));

                    if (raw.length === 3) matieres = raw;

                }

                if (tempsMatieresMatch) {

                    const raw = tempsMatieresMatch[1].split(',').map(s => parseFloat(s.trim()));

                    if (raw.every(v => !isNaN(v)) && raw.length === 3) tempsMatieres = raw;

                }
